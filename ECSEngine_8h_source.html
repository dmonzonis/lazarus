200 template <typename System, typename... EventTypes>
201 void ECSEngine::add_system()
202 {
203  if (!verify_listener<System, EventTypes...>())
204  throw __lz::LazarusException(
205  "System is not a listener to some of the event types");
206 
207  // Put all event types in a vector
208  std::vector<std::type_index> types{__lz::get_type_index<EventTypes>()...};
209 
210  // Search for the system, and if it exists, use it
211  std::shared_ptr<System> new_system = find_system<System>();
212  // If the system does not exist, create a new one
213  if (!new_system)
214  new_system = std::make_shared<System>();
215 
216  for (std::type_index type_id : types)
217  {
218  auto found = subscribers.find(type_id);
219  if (found == subscribers.end())
220  {
221  // No subscribers to this type of event yet, create vector
222  std::vector<std::shared_ptr<__lz::BaseEventListener>> vec;
223  vec.push_back(new_system);
224  subscribers[type_id] = vec;
225  }
226  else
227  {
228  // There already exists a list of subscribers to this event type
229  if (!is_listener<System>(type_id))
230  found->second.push_back(new_system);
231  }
232  }
233 }
